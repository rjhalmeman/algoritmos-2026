Exemplos de Uso de String com Acumulador

Exemplo 1: Concatenar Números


let acumulador = "";
for (let i = 1; i <= 5; i++) {
    acumulador += i + ", ";
}
acumulador = acumulador.slice(0, -2); // Remove última vírgula
console.log(acumulador); // 1, 2, 3, 4, 5

Exemplo 2: Construir Frase com Lista de Nomes


let nomes = ["Ana", "Bruno", "Carlos", "Diana"];
let frase = "Participantes: ";
for (let nome of nomes) {
    frase += nome + ", ";
}
frase = frase.slice(0, -2) + ".";
console.log(frase); // Participantes: Ana, Bruno, Carlos, Diana.

Exemplo 3: Acumular HTML


let html = "";
let frutas = ["Maçã", "Banana", "Laranja"];
for (let fruta of frutas) {
    html += "<li>" + fruta + "</li>";
}
console.log(html);
// <li>Maçã</li><li>Banana</li><li>Laranja</li>

Exemplo 4: Inverter String com Acumulador


let palavra = "algoritmo";
let invertida = "";
for (let i = palavra.length - 1; i >= 0; i--) {
    invertida += palavra[i];
}
console.log(invertida); // omtiroglA

Exemplo 5: Contar Ocorrências e Acumular


let texto = "abaracadabra";
let resultado = "";
let contador = 0;
for (let letra of texto) {
    if (letra === 'a') {
        contador++;
    }
    resultado += letra;
}
console.log("Texto: " + resultado); // abaracadabra
console.log("Letra 'a' aparece: " + contador + " vezes"); // 5 vezes
